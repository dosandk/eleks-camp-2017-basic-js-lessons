return res;
};

function ucFirst(str)
{
    let a = str[0].toUpperCase() + str.slice(1);
    return a;
}
ucFirst("ivan");

function truncate(str, maxlength) {
let s = str; 
let len = maxlength;
if (s.length > len) {
let cutString = s.substring(0, len);
let res = cutString + "...";
return res;
}
else {
return s;
}
}
truncate("long word",6);
